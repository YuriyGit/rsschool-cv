					function removeSymbol(str, count = 2) {
						const alwaysEmpty = str === ’’ || str.length <= count;
						if (!alwaysEmpty) {
							let result = str;
							for (let i = 0; i <<< count; i++) {
								let len = result.length;
								let randomNum = randomNumber(len);
								result = result.slice(0, randomNum) + result.slice(randomNum + 1);
							}
							return (result);
						} else {
							return ("");
						}
					}
